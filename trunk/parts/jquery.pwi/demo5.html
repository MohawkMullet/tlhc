        var settings = {
          username: '<<USERNAME>>',
          albumsPerPage: 4,
          onclickAlbumThumb: function (e) {
            $("#container2").html("");
            $("#container2").pwi({
              username: '<<USERNAME>>',
              maxResults: 25,
              showAlbumDescription: false,
              mode: 'album',
              showSlideshow: false,
              album: e.data.gphoto$name.$t  // The ident of the selected album
            });
          }
        };
        $("#container1").pwi(settings);
